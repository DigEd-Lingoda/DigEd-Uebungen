                    const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/${MODEL_NAME}:generateContent?key=${apiKey}`, {

                        method: 'POST',

                        headers: { 'Content-Type': 'application/json' },

                        body: JSON.stringify(payload)

                    });

                    if (!response.ok) throw new Error('API failed');

                    const data = await response.json();

                    return data.candidates?.[0]?.content?.parts?.[0]?.text || "Ich weiß gerade nicht weiter...";

                } catch (err) {

                    if (retries === maxRetries) throw err;

                    await new Promise(r => setTimeout(r, Math.pow(2, retries) * 1000));

                    retries++;

                }

            }

        }

        async function handleSendMessage(e) {

            e.preventDefault();

            const input = document.getElementById('chat-input');

            const text = input.value.trim();

            if (!text || state.isTyping || state.isConversationEnded) return;

            state.messages.push({ role: 'user', text });

            state.isTyping = true;

            state.exchangeCount++;

            input.value = '';

            render();

            try {

                const aiResponse = await fetchAIResponse(state.messages);

                state.messages.push({ role: 'ai', text: aiResponse });

                if (state.exchangeCount >= 11) {

                    state.isConversationEnded = true;

                }

            } catch (err) {

                console.error(err);

                state.messages.push({ role: 'ai', text: "Entschuldigung, es gab einen technischen Fehler." });

            } finally {

                state.isTyping = false;

                render();

            }

        }
